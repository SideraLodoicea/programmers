# 같은 숫자는 싫어

```java
public class Solution {
    public int[] solution(int []arr) {
        ArrayList<Integer> answer = new ArrayList<>();

        answer.add(arr[0]);
        for(int i=1; i<arr.length; i++) {
            if(arr[i]!=arr[i-1]) {
                answer.add(arr[i]);
            }
        }

        int[] ret = new int[answer.size()];
        for (int i=0; i<ret.length; i++) {
            ret[i] = answer.get(i).intValue();
        }
        return ret;
    }
}
```
